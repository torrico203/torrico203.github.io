C. 유료 결제 시스템 (개발 중 — Xsolla + SendGrid)

// Firebase Functions - 결제 검증 + 메일 발송
exports.onPaymentComplete = functions.https.onRequest(async (req, res) => {
  // 1. Xsolla 웹훅 서명 검증
  const isValid = verifyXsollaSignature(req);
  if (!isValid) return res.status(400).send('Invalid signature');

  const { uid, payContentId, lang } = req.body;

  // 2. 구매 내역 Firebase 저장
  await db.ref(`users/${uid}/purchases/${payContentId}`).set({
    purchasedAt: Date.now(),
    lang,
    resendCount: 0
  });

  // 3. 언어별 PDF 링크 조회 후 SendGrid 메일 발송
  const pdfUrl = await getPdfUrl(payContentId, lang);
  await sendPurchaseMail(userEmail, lang, pdfUrl);

  res.status(200).send('OK');
});

// 마이페이지 재발송 (최대 3회)
exports.resendMail = functions.https.onCall(async (data, context) => {
  const { payContentId, lang } = data;
  const uid = context.auth.uid;

  const purchase = await db.ref(`users/${uid}/purchases/${payContentId}`).get();
  if (purchase.val().resendCount >= 3) throw new Error('재발송 한도 초과');

  const pdfUrl = await getPdfUrl(payContentId, lang);
  await sendPurchaseMail(userEmail, lang, pdfUrl);

  await db.ref(`users/${uid}/purchases/${payContentId}/resendCount`)
    .set(purchase.val().resendCount + 1);
});
✅ 설계 포인트:
• Xsolla 커스텀 UI 결제 + 웹훅 서명 검증
• 언어별 PDF 분기 (ko/en/ja) + 재발송 3회 제한
• Firebase Functions 서버리스 백엔드로 별도 서버 없이 처리

async function loadLikeCounts() {
  const db = getDatabase();
  const likesRef = ref(db, 'games'); // 최상위 노드
  
  const snapshot = await get(likesRef);
  const allGames = snapshot.val();
  
  // 각 게임 카드에 분배
  Object.entries(allGames).forEach(([gameId, data]) => {
    const likeCount = Object.keys(data.likes || {}).length;
    document.querySelector(`#game-${gameId} .like-count`)
      .textContent = likeCount;
  });
}

C. 자체 광고 매니징 시스템 설계

AdManager 아키텍처

// games/commonAd.js
class AdManager {
  constructor() {
    this.adNetwork = this.detectAdNetwork();
    this.userCountry = this.getUserTimezone();
  }

  // 사용자 국가별 광고 분기
  detectAdNetwork() {
    const timezone = Intl.DateTimeFormat().resolvedOptions().timeZone;
    
    if (timezone.includes('Asia/Seoul')) {
      return 'KakaoAdFit'; // 한국
    } else if (timezone.includes('America')) {
      return 'GoogleAdMob'; // 미국
    }
    return 'DefaultAd';
  }

  // 비동기 광고 호출 + 콜백 제어
  async showAd(timing, onCompleteCallback) {
    try {
      // Pre-roll: 게임 시작 전
      // Mid-roll: 플레이 중
      // Post-roll: 결과창
      
      console.log(`[AdManager] Showing ${timing} ad...`);
      
      // 게임 일시정지
      window.gameInstance?.scene.pause();
      
      // 광고 표시 (비동기)
      await this.adNetwork.show();
      
      // 광고 종료 후 게임 재개
      window.gameInstance?.scene.resume();
      
      // 콜백 실행 (예: 결과 화면 표시)
      if (onCompleteCallback) {
        onCompleteCallback();
      }
      
    } catch (error) {
      console.error('Ad failed:', error);
      // 광고 실패해도 게임은 계속
      onCompleteCallback?.();
    }
  }
}

// 게임에서 사용
const adManager = new AdManager();

// 게임 시작 전
await adManager.showAd('pre-roll', () => {
  startGame();
});

// 게임 종료 후
await adManager.showAd('post-roll', () => {
  showResultScreen();
});

캔버스 기반 물리 엔진 & 상태 관리

// games/friedRice/game.js
class FriedRiceGame extends Phaser.Scene {
  constructor() {
    super('FriedRiceGame');
    this.gameState = 'READY'; // READY, PLAYING, RESULT
  }

  preload() {
    // Spine 애니메이션 로드
    this.load.spine('chef', 
      'assets/spine/chef.json',
      'assets/spine/chef.atlas'
    );
    
    // 게임 에셋
    this.load.image('rice', 'assets/rice.png');
    this.load.image('pan', 'assets/pan.png');
  }

  create() {
    // Spine 캐릭터 생성
    this.chef = this.add.spine(400, 300, 'chef');
    this.chef.play('idle', true);
    
    // 물리 엔진 활용
    this.physics.world.gravity.y = 200;
    
    // 상태 관리
    this.input.on('pointerdown', () => {
      if (this.gameState === 'READY') {
        this.startGame();
      } else if (this.gameState === 'PLAYING') {
        this.flipRice();
      }
    });
  }

  startGame() {
    this.gameState = 'PLAYING';
    this.chef.play('cooking', true);
    this.time.delayedCall(30000, this.endGame, [], this);
  }

  flipRice() {
    // 애니메이션 부드러운 전환
    this.chef.setMix('cooking', 'flip', 0.2);
    this.chef.play('flip', false);
    
    this.chef.on('complete', () => {
      this.chef.play('cooking', true);
    });
    
    // 점수 증가
    this.score += 10;
  }

  endGame() {
    this.gameState = 'RESULT';
    this.chef.play('tired', false);
    
    // 광고 표시 후 결과 화면
    adManager.showAd('post-roll', () => {
      this.scene.start('ResultScene', { score: this.score });
    });
  }
}

• 물리 엔진 & 상태 관리 활용

통일된 포털 규격 프리로더

// commandLoading.js
class PortalLoader {
  constructor(gameName) {
    this.gameName = gameName;
    this.progress = 0;
    this.createLoader();
  }

  createLoader() {
    // 포털 통일 UI
    const loaderHTML = `
      <div class="portal-loader">
        <div class="game-title">${this.gameName}</div>
        <div class="progress-bar">
          <div class="fill" id="progress-fill"></div>
        </div>
        <div class="loading-text">Loading...</div>
      </div>
    `;
    
    document.body.insertAdjacentHTML('beforeend', loaderHTML);
  }

  updateProgress(percent) {
    this.progress = percent;
    document.getElementById('progress-fill').style.width = percent + '%';
    
    if (percent >= 100) {
      setTimeout(() => this.hide(), 500);
    }
  }

  hide() {
    document.querySelector('.portal-loader').style.opacity = '0';
    setTimeout(() => {
      document.querySelector('.portal-loader').remove();
    }, 300);
  }
}

// 게임에서 사용
const loader = new PortalLoader('볶음밥 게임');

scene.load.on('progress', (value) => {
  loader.updateProgress(value * 100);
});

  5. 크로스 플랫폼 + 글로벌 대응
    PWA로 iOS/Android 앱 경험 제공, 한/영/일 3개국어 자동 감지, Cloudflare CDN으로 전 세계 배포
